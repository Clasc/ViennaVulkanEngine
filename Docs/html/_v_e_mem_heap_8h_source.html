179  VeMemBlock* pPrev = nullptr;
180  for (VeIndex current_block = 0; current_block < m_pool.size(); ) { //go through all blocks linearly
181  VeMemBlock* pCurrent = getPtr(current_block); //pointer to current block
182  VeIndex next_block = current_block + pCurrent->m_size + sizeof(VeMemBlock); //index of next block
183 
184  if (pCurrent->m_free) { //is current block free?

187  m_first_free = current_block;
188  }
189  else if (pPrev != nullptr) { //if not first block, relink with previous free block
190  pPrev->m_next = current_block;
191  }
192

196  VeMemBlock* pNext = getPtr(next_block);
197  if (pNext->m_free) { //is the next block also free?
198  pCurrent->m_size += pNext->m_size + sizeof(VeMemBlock); //if yes then merge it into the current block
199  m_free_bytes += sizeof(VeMemBlock); //just got more free memory because one less block header
200  pCurrent = pPrev; //make sure that the iterator goes to the current block again
201  next_block = current_block; //iterator will stay on this block
202  }
203  }
204  pPrev = pCurrent; //pointer to previous block (or same if there was a merge)
205  }
